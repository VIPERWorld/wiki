## x!=x

> * http://blog.chinaunix.net/uid-23629988-id-3126229.html
> * http://blog.chinaunix.net/uid-23629988-id-3149291.html

```c
#include <stdlib.h>
#include <stdio.h>
#include <string.h>

int main(void)
{
    float x;
    int a = 1/0.0;//inf
    int a = 0/0.0;//nan
    memcpy(&x, &a, sizeof(x));
    if (x == x) {
        printf("Equal\n");
    } else {
        printf("Not equal\n");
    }

    if (x >= 0.0) {
        printf("x(%f) >= 0\n", x);
    } else if (x < 0.0) {
        printf("x(%f) < 0\n", x);
    } else {
        printf("Surprise x(%f)!!!\n", x);
    }


    return 0;
}
```
非法的另一种来源
```c
#include <stdlib.h>
#include <stdio.h>

int main(void)
{
    float x;

    while (1) {
        scanf("%f", &x);
        printf("x is %f\n", x);
    }

    return 0;
}
```
